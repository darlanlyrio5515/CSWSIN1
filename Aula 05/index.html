    import pandas as pd

    def remove_outlier(df_old):
        # Calculate the IQR for each numerical column
        Q1 = df_old.quantile(0.25)
        Q3 = df_old.quantile(0.75)
        IQR = Q3 - Q1
    
        # Define the limits for the outliers
        lower_bound = Q1 - 1.5 * IQR
        upper_bound = Q3 + 1.5 * IQR
    
        # Filter the dataframe to remove rows with outliers
        df_filtered = df_old[~((df_old < lower_bound) | (df_old > upper_bound)).any(axis=1)]
    
        return df_filtered
